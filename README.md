## Install

```bash
npm install --save requirevsimport
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'requirevsimport'
import 'requirevsimport/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
